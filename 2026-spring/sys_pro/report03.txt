2. 練習問題(301)

2.1 プログラムの説明

以下に，read システムコールと write システムコールを
用いてファイルをコピーするプログラムを示す。
03.html に示されている filecopy-syscall.c をもとにし，
1文字を読み書きする char c の代わりに，大きさ N の
文字配列 buf を用いるように変更した。
コピー元ファイルは src，コピー先ファイルは dst である。
このプログラムでは，通常は N を BUFSIZ としている。
他の N で実験するときは，コンパイル時に -DN=1 や
-DN=1048576 を指定して再コンパイルした。

read は指定したバッファサイズより小さい値を返すことが
あるため，実際に read が返したバイト数だけ write する。

301.c

#include <stdio.h>
#include <stdlib.h>
#include <fcntl.h>
#include <unistd.h>

#ifndef N
#define N BUFSIZ
#endif

int main(void) {
  char buf[N];
  int fps, fpd;
  int count;

  fps = open("src", O_RDONLY);
  if (fps < 0) {
    perror("open: src");
    exit(1);
  }

  fpd = open("dst", O_WRONLY | O_CREAT | O_TRUNC, 0644);
  if (fpd < 0) {
    perror("open: dst");
    close(fps);
    exit(1);
  }

  while ((count = read(fps, buf, N)) > 0) {
    if (write(fpd, buf, count) < 0) {
      perror("write");
      close(fpd);
      close(fps);
      exit(1);
    }
  }

  if (count < 0) {
    perror("read");
    close(fpd);
    close(fps);
    exit(1);
  }

  close(fpd);
  close(fps);

  return 0;
}

2.3 考察

BUFSIZ と 1048576 バイトの場合は，この大きさの入力ではでは 0.00 秒になった。
一方で，1 バイト単位の場合は約 1 秒かかった。
このことからシステムコールを多数回呼び出すことによる
オーバーヘッドが実行時間に大きく影響すると考えられる。

3. 練習問題(303)

3.1 プログラムの説明

以下に，mmap を用いて head コマンドに似た処理を行う
プログラムを示す。第1引数でファイル名を指定し，
第2引数で表示する行数を指定する。ファイル全体を mmap で
メモリに対応づけ，先頭から改行文字を数えることで
出力する範囲を決める。

303_head.c

#include <fcntl.h>
#include <stdio.h>
#include <stdlib.h>
#include <sys/mman.h>
#include <sys/stat.h>
#include <unistd.h>

static size_t map_length(size_t file_size) {
  long page_size = sysconf(_SC_PAGE_SIZE);

  if (page_size <= 0) {
    perror("sysconf");
    exit(1);
  }
  return ((file_size + (size_t)page_size - 1) / (size_t)page_size) *
         (size_t)page_size;
}

int main(int argc, char *argv[]) {
  int fd, err;
  struct stat fs;
  char *p, *p0, *end;
  size_t maplen;
  int line_count, cnt;

  if (argc != 3) {
    exit(1);
  }
  line_count = atoi(argv[2]);

  fd = open(argv[1], O_RDONLY);
  if (fd < 0) {
    perror(argv[1]);
    exit(-1);
  }

  if (fstat(fd, &fs) < 0) {
    perror("fstat");
    exit(-1);
  }

  if (line_count == 0 || fs.st_size == 0) {
    close(fd);
    return 0;
  }

  maplen = map_length((size_t)fs.st_size);

  p = p0 = (char *)mmap(NULL, maplen, PROT_READ, MAP_PRIVATE, fd, 0);
  if (p == MAP_FAILED) {
    perror("mmap");
    exit(-1);
  }

  end = p0 + fs.st_size;
  cnt = 0;

  while (p < end && cnt < line_count) {
    if (*p++ == '\n') {
      cnt++;
    }
  }

  if (write(STDOUT_FILENO, p0, (size_t)(p - p0)) < 0) {
    perror("write");
    exit(-1);
  }

  err = munmap(p0, maplen);
  if (err) {
    perror("munmap");
    exit(-1);
  }
  close(fd);

  return 0;
}

違う点は，ファイル末尾から先頭方向に改行文字を数え出力する部分である.

3.3 考察

どちらのプログラムでも，ファイルサイズが 0 の場合と
表示行数が 0 の場合は mmap を呼ばずに正常終了する。
これは，長さ 0 の mmap が失敗するためである。
また，mmap の第2引数はページサイズの倍数になるように
map_length 関数で切り上げている。
